fname := main

cat:
	cat Makefile

c:
	dmd -de -w ${fname}.d

cr:
	dmd -de -w -run ${fname}.d

r:
	./${fname}

release:
	dmd -release -O ${fname}.d

test:
	dmd -unittest -w -checkaction=context -run ${fname}.d

cov:
	dmd -cov -unittest -w -checkaction=context -run ${fname}.d
	micro ${fname}.lst

profile:
	dmd -profile -g -run ${fname}.d
	echo "see trace.log"

clean:
	rm ${fname}
